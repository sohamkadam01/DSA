class Solution {
    public int slove (int energy[], int idx, int dp[], int k){
        if(idx >= energy.length){
            return 0;
        }
        if(dp[idx] != -1) return dp[idx];

        int total = energy[idx] + slove(energy, idx + k,dp, k);
        dp[idx] = total;
        return dp[idx];
    }
    public int maximumEnergy(int[] energy, int k) {
        int dp[] = new int [energy.length+1];
        Arrays.fill(dp, -1);
        int totalMax = Integer.MIN_VALUE;
        for(int i = 0; i<energy.length;i++){
            int max = slove(energy, i, dp, k);
            if(max > totalMax){
                totalMax = Math.max(max, totalMax);
            }
        }
            return totalMax;
    }
}
