class Solution {
    public int minCost(String colors, int[] neededTime) {
        int minTime = 0;
        int prevIdx = 0;

        for (int i = 1; i < colors.length(); i++) {
            if (colors.charAt(i) == colors.charAt(prevIdx)) {
                if (neededTime[i] > neededTime[prevIdx]) {
                    minTime += neededTime[prevIdx];
                    prevIdx = i;
                } else {
                    minTime += neededTime[i];
                }
            } else {
                prevIdx = i;
            }
        }
        return minTime;
    }
}
