class Solution {
    public boolean isAnagram(String s, String t) {
        if(s.length() != t.length()) return false;
        s.trim();
        t.trim();
     char ch1 [] = s.toCharArray();
     char ch2 [] = t.toCharArray();
     int idx = 0;
     for(char ch : s.toCharArray()){
        ch1[idx++] = ch;
     }
     idx = 0;
     for(char ch : t.toCharArray()){
        ch2[idx++] = ch;
     }
     
     Arrays.sort(ch1);
     Arrays.sort(ch2);

     for(int i = 0; i<ch1.length; i++){
        if(ch1[i] != ch2[i]){
            return false;
        }
     }
     return true;
    }
}
