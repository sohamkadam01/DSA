class Solution {
    public int totalMoney(int n) {
        int total = 0;
        int weekStart = 1; 
        int day = 0;       

        for (int i = 1; i <= n; i++) {
            total += weekStart + day;
            day++;

            if (day == 7) {  
                day = 0;
                weekStart++;
            }
        }

        return total;
    }
}
